#### 3. 创建生成路由 (4 分钟)

**文件**: `backend/routes/generate.js`

**核心代码**:
```javascript
const express = require('express');
const router = express.Router();
const { generatePitch } = require('../services/bailianService');

router.post('/', async (req, res) => {
  try {
    const { project_idea, target_user, progress, pitch_duration, highlight } = req.body;
    
    // 验证输入
    if (!project_idea) {
      return res.status(400).json({
        success: false,
        error: { code: 'INVALID_INPUT', message: '项目想法不能为空' }
      });
    }
    
    // 调用百炼服务
    const result = await generatePitch({
      project_idea,
      target_user,
      progress,
      pitch_duration,
      highlight
    });
    
    res.json({
      success: true,
      data: { result },
      message: '生成成功'
    });
  } catch (error) {
    console.error('生成失败:', error);
    res.status(500).json({
      success: false,
      error: { code: 'INTERNAL_ERROR', message: '生成失败，请稍后重试' }
    });
  }
});

module.exports = router;
```

#### 4. 创建百炼服务 (3 分钟)

**文件**: `backend/services/bailianService.js`

**核心代码**:
```javascript
const axios = require('axios');

/**
 * 调用百炼工作流 API
 */
async function generatePitch(input) {
  const response = await axios.post(
    `https://${process.env.BAILIAN_REGION}.bailian.aliyuncs.com/api/v1/workflows/${process.env.WORKFLOW_ID}/run`,
    {
      input: input
    },
    {
      headers: {
        'Authorization': `Bearer ${process.env.API_KEY}`,
        'Content-Type': 'application/json'
      },
      timeout: 30000 // 30 秒超时
    }
  );
  
  return response.data.output.result;
}

module.exports = { generatePitch };
```
